% ==================================================================
% 1. CAPTURE DATA 
% ==================================================================
\newcommand{\myauthors}{%
$for(by-author)$%
\textbf{$it.name.literal$}%
$if(it.orcid)$~\orcidlink{$it.orcid$}$endif$%
\textsuperscript{$for(it.affiliations)$$it.number$$sep$,$endfor$}%
$if(it.attributes.corresponding)$\textsuperscript{*}$endif$%
$if(it.acknowledgements)$\thanks{$it.acknowledgements$}$endif$%
$sep$, %
$endfor$%
}

\newcommand{\myaffiliations}{%
$for(by-affiliation)$%
\textsuperscript{$it.number$} $it.name$%
$sep$\\ %
$endfor$%
}

\newcommand{\myemail}{%
$for(by-author)$%
$if(it.attributes.corresponding)$%
\textsuperscript{*}Correspondence: \texttt{$it.email$}%
$endif$%
$endfor$%
}

\makeatletter
\renewcommand{\maketitle}{%
  \newpage
  \null
  \vskip 1em%
  % Use a Group with \raggedright instead of 'flushleft' environment
  % This ensures it aligns EXACTLY with the body text margins.
  {\raggedright
    \let \footnote \thanks
    \renewcommand{\thefootnote}{\fnsymbol{footnote}}
    % TITLE
    {\LARGE \bfseries \sffamily \@title \par}%
    \vskip 1em%
    % AUTHORS
    {\large
      \lineskip .5em%
      \noindent\myauthors\par}% 
    % AFFILIATIONS
    \vskip 1em%
    {\footnotesize \myaffiliations \par}%
    % EMAIL
    \vskip 0.5em%
    {\footnotesize \myemail \par}%
  }% End group
  \par
  \vskip 1em}
\makeatother
